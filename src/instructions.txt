 ________________________________________________
|                                                |
|          Pre-processor instructions            |
|________________________________________________|
include external *.jasm file

.inc "[name].jasm"

_________________________________________________
define a name for a constant value

.equ [name] [value]

_________________________________________________
define a label name to jump to

[name]:

 ________________________________________________
|                                                |
|                Assembler code                  |
|________________________________________________|

		SYSTEM INSTRUCTIONS
_________________________________________________
no-operation 

syntax:
	nop	

_________________________________________________
interrupt program

syntax:
	int <imm>
	
_________________________________________________

		ARITHMETIC INSTRUCTIONS
_________________________________________________
addition

syntax:
	add <reg>, <reg>
	add <reg>, <imm>

_________________________________________________
subtraction 

syntax:
	sub <reg>, <reg>
	sub <reg>, <imm>

_________________________________________________
compare 

syntax:
	cmp <reg>, <reg>
	cmp <reg>, <imm>

_________________________________________________
bit-shift left

syntax:
	shl <reg>

_________________________________________________
bit-shift right

syntax:
	shr <reg>

_________________________________________________
negation 

syntax:
	neg <reg>

_________________________________________________
increment register 

syntax:
	inc <reg>

_________________________________________________
decrement register 

syntax:
	dec <reg>

_________________________________________________

		JUMP INSTRUCTIONS
_________________________________________________
jump 

syntax:
	jmp <label>

_________________________________________________
jump if equal, or jump if zero

syntax:
	je <label>

_________________________________________________
jump if not equal, or jump if not zero

syntax:
	jne <label>

_________________________________________________
jump if greater 

syntax:
	jg <label>

_________________________________________________
jump if greater or equal 

syntax:
	jge <label>

_________________________________________________
jump if less 

syntax:
	jl <label>

_________________________________________________
jump if less or equal 

syntax:
	jle <label>

_________________________________________________
call label 

syntax:
	call <label>

_________________________________________________
return to previous call 

syntax:
	ret	

_________________________________________________

		DATA TRANSFER INSTRUCTIONS
_________________________________________________
load register

syntax:
	load <reg>, <reg>
	load <reg>, <imm>

_________________________________________________
push to stack

syntax:
	push <reg>
	push <imm>

_________________________________________________
pop from stack 

syntax:
	pop <reg>

_________________________________________________
store to ram

syntax:
	sto @<reg>,<reg>
	sto @<imm>,<reg>
	
_________________________________________________
read ram

syntax:
	read <reg>, @<reg>
	read <reg>, @<imm>
